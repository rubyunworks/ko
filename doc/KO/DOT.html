def report(entry)
  case entry[:type]
  when 'header'
    @_rec = []
    @_time = Time.now
  when 'case'
  when 'note'
  when 'test'
    case entry[:status]
    when 'pass'
      print '.'
    when 'fail'
      @_rec << entry
      print 'F'
    when 'error'
      @_rec << entry
      print 'E'
    end
  when 'footer'
    puts
    @_rec.each do |e|
      puts
      puts e[:status].upcase + ' ' + e[:label]
      puts e[:message]
      e[:snippet].each do |h|
        l, s = h.keys.first, h.values.first
        if l.to_i == e[:line]
          puts " > %5d: %s" % [l.to_i,s]
        else
          puts " %7d: %s" % [l.to_i,s]
        end
      end
    end
    puts
    puts "Finished in #{Time.now - @_time} seconds."
    puts
    sums = %w{pass fail error pending}.map{ |w| entry[:tally][w] }
    puts "#{entry[:count]} tests: %s pass, %s fail, %s err, %s pending" % sums
  end
end
