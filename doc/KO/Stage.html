def clear
  raise "unsafe test stage directory -- #{Dir.pwd}" unless /#{Dir.tmpdir}/ =~ Dir.pwd
  Dir['*'].each do |path|
    FileUtils.rm_r(path)
  end
end

def copy(source_dir)
  test_dir = File.dirname(caller[0])
  stage_clear
  srcdir = File.join(test_dir, source_dir)
  Dir[File.join(srcdir, '*')].each do |path|
    FileUtils.cp_r(path, '.')
  end
end
