def report(entry)
  case entry[:type]
  when 'header'
    puts "#{entry[:range]}"
  when 'case'
    puts "# #{entry[:description]}"
  when 'note'
    puts "# #{entry[:description]}"
  when 'footer'
    puts "#{entry[:range]}"
  when 'test'
    case entry[:status]
    when 'pass'
      puts "ok #{entry[:index]} - #{entry[:label]}"
    when 'fail', 'error'
      puts "not ok #{entry[:index]} - #{entry[:label]}"
      puts "  ---"
      puts "  description: #{entry[:message]}"
      puts "  file: #{entry[:file]}"
      puts "  line: #{entry[:line]}"
      puts "  raw_test: #{entry[:source]}"
      puts "  ..."
    end
  else
    # ?
  end
end
